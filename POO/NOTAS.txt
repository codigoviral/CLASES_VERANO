import mysql.connector

class Cliente:
    def _init_(self, rut, nombre):
        self.rut = rut
        self.nombre = nombre
    def getRut(self):
        return self.rut
    def setRut(self, rut):
        self.rut = rut
    def getNombre(self):
        return self.nombre
    def setNombre(self, nombre):
        self.nombre = nombre

class ServicioCliente:
    def _init_(self, host, usuario, password, baseDatos):
        self.conexion = mysql.connector.connect(
            host = host,
            user = usuario,
            password = password,
            database = baseDatos
        )
        self.cursor = self.conexion.cursor()

    def agregarCliente(self, cliente):
        sql = "INSERT INTO CLIENTE(RUT, NOMBRE) VALUES (%s, %s)"
        val = (cliente.getRut(), cliente.getNombre())
        self.cursor.execute(sql, val)
        self.conexion.commit()
    
    def cerrarConexion(self):
        self.cursor.close()
        self.conexion.close()

cliente = Cliente(1, "a")
servicio = ServicioCliente("localhost", "root", "", "BDCLIENTE")
servicio.agregarCliente(cliente)
servicio.cerrarConexion()
